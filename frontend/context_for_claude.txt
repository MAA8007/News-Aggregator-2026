"use client";

import Image from "next/image";
import { motion } from "framer-motion";
import type { Article } from "@/app/types";

interface NewsCardProps {
  article: Article;
  isHero?: boolean;
  index?: number;
  priority?: boolean;
}

export default function NewsCard({
  article,
  isHero = false,
  index = 0,
  priority = false,
}: NewsCardProps) {
  const hasImage = !!article.image_url;

  const handleClick = () => {
    window.open(article.link, "_blank", "noopener,noreferrer");
  };

  const motionProps = {
    initial: { opacity: 0, y: 20 },
    whileInView: { opacity: 1, y: 0 },
    transition: { duration: 0.45, delay: index * 0.05, ease: "easeOut" },
    viewport: { once: true, margin: "-80px" },
  };

  /* ── State A: Image present ─────────────────────────────────────── */
  if (hasImage) {
    return (
      <motion.div
        {...motionProps}
        onClick={handleClick}
        className="group relative h-full overflow-hidden rounded-3xl cursor-pointer shadow-card hover:shadow-card-hover transition-shadow duration-300"
        style={{ willChange: "transform" }}
      >
        {/* Image with zoom on hover */}
        <motion.div
          whileHover={{ scale: 1.06 }}
          transition={{ duration: 0.4, ease: "easeOut" }}
          className="absolute inset-0"
        >
          <Image
            src={article.image_url!}
            alt={article.title}
            fill
            className="object-cover"
            sizes={isHero ? "(max-width: 768px) 100vw, 50vw" : "(max-width: 768px) 100vw, 25vw"}
            priority={priority}
            unoptimized
          />
        </motion.div>

        {/* Dark gradient for text legibility */}
        <div className="absolute inset-x-0 bottom-0 h-4/5 bg-gradient-to-t from-black/90 via-black/50 to-transparent pointer-events-none" />

        {/* Category badge */}
        {article.category && (
          <div className="absolute top-3 left-3 z-10">
            <span className="font-sans text-xs font-medium px-2.5 py-1 rounded-full bg-black/40 backdrop-blur-sm text-white/85 border border-white/15 tracking-wide">
              {article.category}
            </span>
          </div>
        )}

        {/* Headline + meta at bottom */}
        <div className="absolute inset-x-0 bottom-0 z-10 p-4 sm:p-5">
          <div className="flex items-center gap-2 mb-2 text-white/55 text-xs font-sans">
            <span className="font-medium text-white/70">{article.source}</span>
            {article.pub_date && (
              <>
                <span className="text-white/30">·</span>
                <time>
                  {new Date(article.pub_date).toLocaleDateString("en-US", {
                    month: "short",
                    day: "numeric",
                  })}
                </time>
              </>
            )}
          </div>
          <h2
            className={`font-display font-extrabold leading-tight text-white drop-shadow-sm ${
              isHero
                ? "text-2xl sm:text-4xl line-clamp-3"
                : "text-base sm:text-lg line-clamp-3"
            }`}
          >
            {article.title}
          </h2>
        </div>
      </motion.div>
    );
  }

  /* ── State B: No image — frosted quartz typographic card ────────── */
  const snippet = article.content
    ? article.content.replace(/\s+/g, " ").trim().slice(0, 150)
    : null;

  return (
    <motion.div
      {...motionProps}
      onClick={handleClick}
      className="quartz-card group h-full"
    >
      <div className="h-full flex flex-col p-5 sm:p-6">
        {/* Top row: category + date */}
        <div className="flex items-center justify-between mb-auto">
          {article.category ? (
            <span className="font-sans text-xs font-semibold uppercase tracking-widest text-gray-400">
              {article.category}
            </span>
          ) : (
            <span />
          )}
          {article.pub_date && (
            <time className="font-sans text-xs text-gray-400">
              {new Date(article.pub_date).toLocaleDateString("en-US", {
                month: "short",
                day: "numeric",
              })}
            </time>
          )}
        </div>

        {/* Centred serif headline */}
        <div className="flex-1 flex items-center justify-center py-4">
          <h2
            className={`font-display font-extrabold text-gray-900 text-center leading-snug line-clamp-4 group-hover:text-gray-600 transition-colors duration-200 ${

            {article.title}

"use client";

import { motion } from "framer-motion";
import type { Article } from "@/app/types";
import NewsCard from "./NewsCard";

interface NewsGridProps {
  articles: Article[];
}

/* ── Helpers ──────────────────────────────────────────────────────── */
function countTopImages(articles: Article[]): number {
  return articles.slice(0, 5).filter((a) => !!a.image_url).length;
}

/* ── Broadsheet card (text-only, newspaper column style) ─────────── */
function BroadsheetCard({ article, index }: { article: Article; index: number }) {
  const snippet = article.content
    ? article.content.replace(/\s+/g, " ").trim().slice(0, 200)
    : null;

  return (
    <motion.div
      initial={{ opacity: 0, y: 16 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.4, delay: index * 0.06, ease: "easeOut" }}
      onClick={() => window.open(article.link, "_blank", "noopener,noreferrer")}
      className="break-inside-avoid mb-8 pb-8 border-b border-gray-200 cursor-pointer group"
    >
      <div className="flex items-center gap-2 mb-3">
        {article.category && (
          <span className="font-sans text-[10px] font-semibold uppercase tracking-widest text-gray-400">
            {article.category}
          </span>
        )}
        {article.category && article.pub_date && (
          <span className="text-gray-300 text-xs">·</span>
        )}
        {article.pub_date && (
          <time className="font-sans text-xs text-gray-400">
            {new Date(article.pub_date).toLocaleDateString("en-US", {
              month: "long",
              day: "numeric",
              year: "numeric",
            })}
          </time>
        )}
      </div>

      <h2 className="font-display font-bold text-gray-900 text-xl sm:text-2xl leading-snug mb-3 group-hover:text-gray-500 transition-colors duration-200">
        {article.title}
      </h2>

      {snippet && (
        <p className="font-sans text-sm text-gray-500 leading-relaxed line-clamp-3">
          {snippet}
        </p>
      )}

      <span className="block font-sans text-xs font-medium text-gray-400 mt-3">
        {article.source}
      </span>
    </motion.div>
  );
}

/* ── Broadsheet layout (2-column print newspaper flow) ───────────── */
function BroadsheetLayout({ articles }: { articles: Article[] }) {
  return (
    <div className="max-w-4xl mx-auto">
      {/* Masthead rule */}
      <div className="flex items-center gap-4 mb-10">
        <div className="flex-1 h-px bg-gray-300" />
        <span className="font-sans text-xs font-semibold uppercase tracking-[0.2em] text-gray-400">
          Latest
        </span>
        <div className="flex-1 h-px bg-gray-300" />
      </div>

      {/* Column flow */}
      <div className="sm:columns-2 gap-10">
        {articles.map((article, idx) => (
          <BroadsheetCard key={article.id} article={article} index={idx} />
        ))}
      </div>
    </div>
  );
}

/* ── Bento grid ──────────────────────────────────────────────────── */
function BentoGrid({ articles }: { articles: Article[] }) {
  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-5 auto-rows-[280px]">
      {/* Hero (2×2) */}
      {articles[0] && (
        <div className="lg:col-span-2 lg:row-span-2">
          <NewsCard article={articles[0]} isHero index={0} priority />
        </div>
      )}

      {/* Sidekick 1 (1×1) */}
      {articles[1] && (
        <div className="lg:col-span-1">
          <NewsCard article={articles[1]} index={1} priority />
        </div>
      )}

      {/* Sidekick 2 (1×1) */}
      {articles[2] && (
        <div className="lg:col-span-1">
          <NewsCard article={articles[2]} index={2} priority />
        </div>
      )}

      {/* Tall Highlight (1×2) */}
      {articles[3] && (
        <div className="lg:col-span-1 lg:row-span-2">
          <NewsCard article={articles[3]} index={3} priority />
        </div>
      )}

      {/* Standard feed (1×1) */}
      {articles.slice(4).map((article, idx) => (
        <div key={article.id} className="lg:col-span-1">
          <NewsCard article={article} index={idx + 4} />
        </div>
      ))}
    </div>
  );
}

/* ── Main export ─────────────────────────────────────────────────── */
export default function NewsGrid({ articles }: NewsGridProps) {
  if (articles.length === 0) return null;

  const topImagesCount = countTopImages(articles);

  if (topImagesCount === 0) {
    return <BroadsheetLayout articles={articles} />;
  }

  return <BentoGrid articles={articles} />;
}

"use client";

import type { Filters } from "@/app/types";

const CATEGORY_COLOUR: Record<string, string> = {
  "AI News": "pill-active-cyan",
  "Science & Technology": "pill-active-violet",
  Football: "pill-active-green",
};

function categoryPillClass(category: string, active: boolean): string {
  if (!active) return "pill pill-inactive";
  return `pill ${CATEGORY_COLOUR[category] ?? "pill-active-default"}`;
}

interface FilterBarProps {
  filters: Filters;
  activeCategory: string | null;
  activeSource: string | null;
  onCategoryChange: (cat: string | null) => void;
  onSourceChange: (src: string | null) => void;
}

export default function FilterBar({
  filters,
  activeCategory,
  activeSource,
  onCategoryChange,
  onSourceChange,
}: FilterBarProps) {
  if (filters.categories.length === 0 && filters.sources.length === 0) {
    return null;
  }

  return (
    <div className="sticky top-[65px] z-20 backdrop-blur-md bg-white/75 border-b border-gray-200/60">
      <div className="max-w-[1600px] mx-auto px-4 sm:px-6 lg:px-8">
        <div className="flex items-center gap-2 overflow-x-auto scrollbar-hide py-3">

          {/* Category pills */}
          <button
            className={`pill ${!activeCategory ? "pill-active-default" : "pill-inactive"}`}
            onClick={() => onCategoryChange(null)}
          >
            All
          </button>

          {filters.categories.map((cat) => (
            <button
              key={cat}
              className={categoryPillClass(cat, activeCategory === cat)}
              onClick={() => onCategoryChange(activeCategory === cat ? null : cat)}
            >
              {cat}
            </button>
          ))}

          {/* Divider */}
          {filters.sources.length > 0 && (
            <div className="h-4 w-px bg-gray-300 mx-1 shrink-0" />
          )}

          {/* Source pills */}
          {filters.sources.map((src) => (
            <button
              key={src}
              className={`pill ${activeSource === src ? "pill-active-amber" : "pill-inactive"}`}
              onClick={() => onSourceChange(activeSource === src ? null : src)}
            >
              {src}
            </button>
          ))}
        </div>
      </div>
    </div>
  );
}

import type { Filters, PaginatedArticles } from "@/app/types";

const API_BASE =
  process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:8000";

export async function fetchNews(params: {
  page?: number;
  pageSize?: number;
  category?: string | null;
  source?: string | null;
}): Promise<PaginatedArticles> {
  const { page = 1, pageSize = 12, category, source } = params;
  const url = new URL(`${API_BASE}/api/news`);
  url.searchParams.set("page", String(page));
  url.searchParams.set("page_size", String(pageSize));
  if (category) url.searchParams.set("category", category);
  if (source) url.searchParams.set("source", source);

  const res = await fetch(url.toString(), { cache: "no-store" });
  if (!res.ok) throw new Error(`fetchNews failed: ${res.status}`);
  return res.json();
}

export async function fetchFilters(): Promise<Filters> {
  const res = await fetch(`${API_BASE}/api/news/filters`, { cache: "no-store" });
  if (!res.ok) throw new Error(`fetchFilters failed: ${res.status}`);
  return res.json();
}

"use client";

import { useEffect, useState } from "react";

import FilterBar from "@/app/components/FilterBar";
import Header from "@/app/components/Header";
import NewsGrid from "@/app/components/NewsGrid";
import SkeletonGrid from "@/app/components/SkeletonGrid";
import { fetchFilters, fetchNews } from "@/app/lib/api";
import type { Article, Filters } from "@/app/types";

const PAGE_SIZE = 12;

export default function HomePage() {
  const [articles, setArticles] = useState<Article[]>([]);
  const [total, setTotal] = useState(0);
  const [page, setPage] = useState(1);
  const [initialLoading, setInitialLoading] = useState(true);
  const [loadingMore, setLoadingMore] = useState(false);
  const [category, setCategory] = useState<string | null>(null);
  const [source, setSource] = useState<string | null>(null);
  const [filters, setFilters] = useState<Filters>({ categories: [], sources: [] });
  const [error, setError] = useState<string | null>(null);

  // Load filter options once
  useEffect(() => {
    fetchFilters()
      .then(setFilters)
      .catch(() => {/* non-fatal */});
  }, []);

  // Reset + fetch page 1 whenever a filter changes
  useEffect(() => {
    let active = true;
    setInitialLoading(true);
    setError(null);
    setPage(1);

    fetchNews({ page: 1, pageSize: PAGE_SIZE, category, source })
      .then((data) => {
        if (!active) return;
        setArticles(data.items);
        setTotal(data.total);
        setInitialLoading(false);
      })
      .catch(() => {
        if (!active) return;
        setError("Could not reach the API. Is the backend running on port 8000?");
        setInitialLoading(false);
      });

    return () => { active = false; };
  }, [category, source]);

  // Fetch next pages (page > 1)
  useEffect(() => {
    if (page === 1) return;
    let active = true;
    setLoadingMore(true);

    fetchNews({ page, pageSize: PAGE_SIZE, category, source })
      .then((data) => {
        if (!active) return;
        setArticles((prev) => [...prev, ...data.items]);
        setTotal(data.total);
        setLoadingMore(false);
      })
      .catch(() => {
        if (!active) return;
        setLoadingMore(false);
      });

    return () => { active = false; };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [page]);

  const hasMore = articles.length < total;

  return (
    <main className="min-h-screen">
      <div className="relative z-10">
        <Header total={total} />

        <FilterBar
          filters={filters}
          activeCategory={category}
          activeSource={source}
          onCategoryChange={(cat) => setCategory(cat)}
          onSourceChange={(src) => setSource(src)}
        />

        <div className="max-w-[1600px] mx-auto px-4 sm:px-6 lg:px-8 py-8">

          {/* ── Error state ── */}
          {error && (
            <div className="glass rounded-2xl p-10 text-center max-w-lg mx-auto mt-16">
              <p className="font-sans text-red-600 font-medium mb-2">Connection error</p>
              <p className="font-sans text-gray-500 text-sm">{error}</p>
            </div>
          )}

          {/* ── Loading skeletons ── */}
          {!error && initialLoading && <SkeletonGrid count={8} />}

          {/* ── Empty state ── */}
          {!error && !initialLoading && articles.length === 0 && (
            <div className="glass rounded-2xl p-10 text-center max-w-lg mx-auto mt-16">
              <p className="font-sans text-gray-600 font-medium mb-2">No articles yet</p>
              <p className="font-sans text-gray-400 text-sm">
                Trigger a scrape via{" "}
                <code className="bg-gray-100 px-1.5 py-0.5 rounded text-indigo-600 font-sans text-xs">
                  POST /api/scrape
                </code>{" "}
                to populate the feed.
              </p>
            </div>
          )}

          {/* ── Article grid ── */}
          {!error && !initialLoading && articles.length > 0 && (
            <>
              <NewsGrid articles={articles} />

              {/* Load more */}
              {hasMore && (
                <div className="flex justify-center mt-12">
                  <button
                    className="glass-btn px-10 py-3 text-sm font-semibold tracking-wide"
                    onClick={() => setPage((p) => p + 1)}
                    disabled={loadingMore}
                  >
                    {loadingMore ? (
                      <span className="flex items-center gap-2">
                        <span className="w-3.5 h-3.5 rounded-full border-2 border-gray-400 border-t-white animate-spin" />
                        Loading…
                      </span>
                    ) : (
                      `Load more · ${total - articles.length} remaining`
                    )}
                  </button>
                </div>
              )}

              {!hasMore && (
                <p className="font-sans text-center text-gray-400 text-sm mt-12">
                  You&apos;ve reached the end · {total} articles total
                </p>
              )}
            </>
          )}
        </div>
      </div>
    </main>
  );
}
